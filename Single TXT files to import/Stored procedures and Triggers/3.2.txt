DROP PROCEDURE IF EXISTS szkola.etat_pol;
DELIMITER $$
CREATE PROCEDURE etat_pol()
	UPDATE pracownicy
	SET etat = "pó³"
    WHERE pensja < 1700

$$
DELIMITER ;

CALL etat_pol;

CREATE PROCEDURE etat_pelen()

	SET etat = "pe³en"
    WHERE pensja >= 1700
$$
